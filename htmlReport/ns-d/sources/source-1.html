 import com.svantheemsche.appetijt.domain.model.Recipe
 import androidx.compose.foundation.ExperimentalFoundationApi
 
 @OptIn(ExperimentalFoundationApi::class)
 @Composable
 fun RecipeCard(
     recipe: Recipe,
     onClick: () -> Unit,
     onLongClick: () -> Unit,
     modifier: Modifier = Modifier
 ) {
     val sanitizedTitle = remember(recipe.title) { HtmlSanitizer.sanitize(recipe.title) }
 
     OutlinedCard(
         modifier = modifier
             .fillMaxWidth()
             .combinedClickable(
                 onClick = onClick,
                 onLongClick = onLongClick
             ),
         shape = RoundedCornerShape(16.dp),
         colors = CardDefaults.outlinedCardColors(
             containerColor = MaterialTheme.colorScheme.surface,
         ),
         border = BorderStroke(1.5.dp, MaterialTheme.colorScheme.onBackground),
         elevation = CardDefaults.cardElevation(defaultElevation = 0.dp)
     ) {
         Row(
             modifier = Modifier
                 .padding(12.dp)
                 .fillMaxWidth(),
             verticalAlignment = Alignment.CenterVertically
         ) {
             SubcomposeAsyncImage(
                 model = recipe.imageUrl,
                 contentDescription = "Thumbnail for ${recipe.title}",
                 modifier = Modifier
                     .size(64.dp),
                 contentScale = ContentScale.Crop,
                 loading = {
                     Box(
                         modifier = Modifier
                             .fillMaxSize()
                             .background(Color.LightGray)
                     )
                 },
                 error = {
                     Box(
                         modifier = Modifier
                             .fillMaxSize()
                             .background(Color.Gray)
                     )
                 }
             )
             
             Spacer(modifier = Modifier.width(12.dp))
             
             Column {
                 Text(
                     text = sanitizedTitle,
                     style = MaterialTheme.typography.titleMedium,
                     maxLines = 3,
                     overflow = TextOverflow.Ellipsis
                 )
                 recipe.sourceApp?.let {
                     Text(
                         text = it,
                         style = MaterialTheme.typography.bodySmall,
                         color = MaterialTheme.colorScheme.onSurfaceVariant
                     )
                 }
             }
         }
     }
 }
